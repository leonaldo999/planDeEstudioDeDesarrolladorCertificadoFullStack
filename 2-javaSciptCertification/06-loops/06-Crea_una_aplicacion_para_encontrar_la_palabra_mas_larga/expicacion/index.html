📝 Código Explicado

function findLongestWordLength(oracion) {
  let maxLength = 0;
  let currentLength = 0;
  
  // Recorrer cada carácter
  for (let i = 0; i < oracion.length; i++) {
    if (oracion[i] !== ' ') {
      currentLength++;
    } else {
      if (currentLength > maxLength) {
        maxLength = currentLength;
      }
      currentLength = 0;
    }
  }
  
  // Verificar última palabra
  if (currentLength > maxLength) {
    maxLength = currentLength;
  }
  
  return maxLength;
}
